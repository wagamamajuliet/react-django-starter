--프론트엔드 폴더로 접근

cd frontend

--만약 yarn이 없다면, yarn 설치

npm install --global yarn

--yarn 으로 package.json에 있는 라이브러리 모듈 설치

yarn

--yarn build 실행

yarn build

--다시 기본 폴더로 돌아옴

cd..

--파이썬 가상환경 설정

python -m venv myvenv

--파이썬 가상환경 실행

myvenv\Scripts\activate

--장고 설치

pip install django

--데이터베이스 설정

python manage.py migrate

--장고 실행

python manage.py runserver

파이썬리액트 8000포트로 실행됨

localhost:8000
